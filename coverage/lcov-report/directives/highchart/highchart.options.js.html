	generateDates(points, begin, end) {
		const dates = [];
		const results = [];
 
		for (let i = 0; i < points.length; i++) {
			dates.push([
				Date.UTC(
					this.random(begin.getFullYear(), end.getFullYear()),
					this.random(begin.getMonth(), end.getMonth()),
					this.random(begin.getDay(), end.getDay())
				)
			]);
		}
 
		dates.sort((a, b) => {
			return a - b;
		});
 
		for (let i = 0; i < points.length; i++) {
			results.push([dates[i][0], points[i]]);
		}
 
		return results;
	}
 
	random(min, max) {
		return Math.random() * (max - min) + min;
